**Part A: Add dedup to monitor.ts**

Add a dedup module-level Map and `isDuplicate` function near the top of monitor.ts (after imports, before the webhook handler):

```typescript
// Webhook deduplication — safety net for duplicate deliveries
// Primary guard is the message vs message.any event filter (line ~2104)
// This is SECONDARY protection using composite key
const DEDUP_MAX = 200;
const DEDUP_TTL_MS = 5 * 60 * 1000; // 5 minutes

const _dedupEntries = new Map<string, number>(); // key -> timestamp

function isDuplicate(eventType: string, messageId: string | undefined): boolean {
  if (!messageId) return false;
  const key = `${eventType}:${messageId}`;

  // Prune expired entries when approaching max
  if (_dedupEntries.size > DEDUP_MAX) {
    const now = Date.now();
    for (const [k, ts] of _dedupEntries) {
      if (now - ts > DEDUP_TTL_MS) _dedupEntries.delete(k);
    }
  }

  if (_dedupEntries.has(key)) return true;
  _dedupEntries.set(key, Date.now());
  return false;
}
```

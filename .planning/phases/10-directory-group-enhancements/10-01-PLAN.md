      const noNameParts = participants.filter(p => !p.displayName);

        const contact = db.getContact(altJid);
        if (contact?.displayName) {
          db.db.prepare("UPDATE group_participants SET display_name = ? WHERE group_jid = ? AND participant_jid = ?")

2. The button needs to reflect the current dm_allowed state. Check the data returned by `/api/admin/directory?type=newsletter` — does it include dm_allowed or allow_dm state? If not, the API response needs to include it.
3. Change the button to a toggle:
```javascript
var allowBtn = document.createElement("button");
allowBtn.className = isAllowed ? "btn btn-sm btn-success" : "btn btn-sm btn-outline";
allowBtn.textContent = isAllowed ? "Allowed" : "Allow DM";
```

4. On click, toggle the state:
```javascript
allowBtn.addEventListener("click", function() {
  var newState = !isAllowed;
  fetch("/api/admin/directory/" + encodeURIComponent(jid) + "/allow-dm", {
    method: "PUT",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({allow: newState})
  }).then(function(r) {
    if (!r.ok) throw new Error("Status " + r.status);
    isAllowed = newState;
    allowBtn.textContent = newState ? "Allowed" : "Allow DM";
    allowBtn.className = newState ? "btn btn-sm btn-success" : "btn btn-sm btn-outline";
    allowBtn.style.cssText = newState ? "background:#2e7d32; color:#fff;" : "background:transparent; color:#78909c; border:1px solid #546e7a;";
    showToast(newState ? "DM allowed for channel" : "DM revoked for channel", false);
  }).catch(function(err) {
    showToast("Failed: " + err.message, true);
  });
});
```

5. The visual state must persist across page reload — it reads from the API response data, not local state.

2. After a successful role change to `bot_admin` or `manager`, automatically fire two additional API calls:
   a. Enable Allow (group): `PUT /api/admin/directory/group/:groupJid/participants/:participantJid/allow` with `{allow: true}`
   b. Enable Allow DM: `PUT /api/admin/directory/:participantJid/allow-dm` with `{allow: true}`
3. On success of both, show toast: "Role updated. Allow and Allow DM auto-enabled."
4. Update the button visuals inline to reflect the new allow states (change button text/color to "Allowed").
5. When demoting from `bot_admin`/`manager` to `participant`:
   a. Auto-revoke Allow DM: `PUT /api/admin/directory/:participantJid/allow-dm` with `{allow: false}`
   b. Keep group Allow as-is (they're still in the group)
   c. Show toast: "Role updated. Allow DM revoked."
6. Update button visuals to reflect the revoked DM state.

Implementation:
```javascript
function onRoleChange(groupJid, participantJid, newRole, prevRole) {
  // Save role first
  fetch("/api/admin/directory/group/" + encodeURIComponent(groupJid) + "/participants/" + encodeURIComponent(participantJid) + "/role", {
    method: "PUT",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({role: newRole})
  }).then(function(r) {
    if (!r.ok) throw new Error("Role save failed");

    if (newRole === "bot_admin" || newRole === "manager") {
      // Auto-grant
      return Promise.all([
        fetch("/api/admin/directory/group/" + encodeURIComponent(groupJid) + "/participants/" + encodeURIComponent(participantJid) + "/allow", {
          method: "PUT",
          headers: {"Content-Type": "application/json"},
          body: JSON.stringify({allow: true})
        }),
        fetch("/api/admin/directory/" + encodeURIComponent(participantJid) + "/allow-dm", {
          method: "PUT",
          headers: {"Content-Type": "application/json"},
          body: JSON.stringify({allow: true})
        })
      ]).then(function() {
        showToast("Role updated. Allow and Allow DM auto-enabled.", false);
        // Update button visuals...
      });
    } else if ((prevRole === "bot_admin" || prevRole === "manager") && newRole === "participant") {
      // Auto-revoke DM only
      return fetch("/api/admin/directory/" + encodeURIComponent(participantJid) + "/allow-dm", {
        method: "PUT",
        headers: {"Content-Type": "application/json"},
        body: JSON.stringify({allow: false})
      }).then(function() {
        showToast("Role updated. Allow DM revoked.", false);
        // Update button visuals...
      });
    } else {
      showToast("Role updated.", false);
    }
  }).catch(function(err) {
    showToast("Error: " + err.message, true);
  });
}
```

Wire this function to the role dropdown's `onchange` event, replacing the existing handler. Make sure to capture `prevRole` before the change (read from the dropdown's previous selectedIndex or a data attribute).

```typescript
exportConfig: () => {
  // Direct fetch to trigger download — not using request() since we need the blob
  const base = (window as any).__API_BASE ?? '/api/admin';
  return fetch(`${base}/config/export`).then(r => {
    if (!r.ok) throw new Error('Export failed');
    return r.blob();
```

2. **Update handleSave to parse validation errors:**
   In the catch block, check if the error response contains `validation_failed`:
   - If the API returns a 400 with `{ error: "validation_failed", fields: [...] }`, parse the fields array
   - Convert to a `Record<string, string>` keyed by field path (e.g., `"dmFilter.tokenEstimate"` -> `"Expected number..."`)
   - Set `fieldErrors` state with this map
   - Show toast: `"Validation failed: {N} field(s) have errors"` (not generic "Save failed")
   - On successful save, clear `fieldErrors` to empty object

   To make this work, update the `api.updateConfig` call to handle structured errors. The `request()` helper likely throws on non-200. Catch the error, check if it has a `.fields` property (or parse the response body). The cleanest approach: update `handleSave` to use fetch directly or modify the error handling to preserve the response body.

   Recommended approach — modify `handleSave`:
   ```typescript
   async function handleSave() {
     setSaving(true)
     setFieldErrors({})
     try {
       await api.updateConfig(buildPayload())
       setDirty(false)
       toast.success('Settings saved')
     } catch (err: any) {
       // Check for structured validation errors from backend
       if (err?.error === 'validation_failed' && Array.isArray(err?.fields)) {
         const errMap: Record<string, string> = {}
         for (const f of err.fields) errMap[f.path] = f.message
         setFieldErrors(errMap)
         toast.error(`Validation failed: ${err.fields.length} field(s) have errors`)
       } else {
         toast.error(`Save failed: ${err instanceof Error ? err.message : String(err)}`)
       }
     } finally {
       setSaving(false)
     }
   }
   ```

   NOTE: Check how the `request()` helper in api.ts handles non-2xx responses. If it throws a plain Error, you'll need to update it to preserve the JSON body. The simplest fix: in `request()`, when response is not ok, parse the JSON body and throw it as-is (or as a custom error with `.error` and `.fields` properties).

   ```tsx
   function FieldError({ path, errors }: { path: string; errors: Record<string, string> }) {
   ```

   ```typescript
     try {
       const blob = await api.exportConfig()

       a.download = 'openclaw-config.json'
   ```

   ```typescript
   async function handleImport(e: React.ChangeEvent<HTMLInputElement>) {
     const file = e.target.files?.[0]
     if (!file) return
     try {
       const text = await file.text()
       const parsed = JSON.parse(text)
       await api.importConfig(parsed)
       toast.success('Config imported successfully')
       // Reload config from server
       const resp = await api.getConfig()
       setConfig(resp.waha)
       setDirty(false)
       setFieldErrors({})
     } catch (err: any) {
       if (err?.error === 'validation_failed' && Array.isArray(err?.fields)) {
         const errMap: Record<string, string> = {}
         for (const f of err.fields) errMap[f.path] = f.message
         setFieldErrors(errMap)
         toast.error(`Import validation failed: ${err.fields.length} field(s) have errors`)
       } else {
         toast.error(`Import failed: ${err instanceof Error ? err.message : String(err)}`)
       }
     } finally {
       // Reset file input so same file can be re-selected
       if (fileInputRef.current) fileInputRef.current.value = ''
     }
   }
   ```

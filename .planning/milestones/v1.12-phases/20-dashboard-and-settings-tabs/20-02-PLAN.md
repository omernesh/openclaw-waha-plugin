**Part 4: godModeSuperUsers serialization -- CRITICAL**

The API stores godModeSuperUsers as `Array<{ identifier: string }>` but TagInput works with `string[]`. Create two helper functions:

```typescript
// Extract JIDs from config for TagInput display
function godModeJids(filter: 'dm' | 'group'): string[] {
  const users = filter === 'dm'
    ? config?.dmFilter?.godModeSuperUsers
    : config?.groupFilter?.godModeSuperUsers
  return (users ?? []).map(u => u.identifier)
}

// Convert TagInput string[] back to API format for save
function updateGodModeUsers(filter: 'dm' | 'group', jids: string[]) {
  const key = filter === 'dm' ? 'dmFilter' : 'groupFilter'
  updateConfig(`${key}.godModeSuperUsers`, jids.map(jid => ({ identifier: jid })))
}
```

Update `buildPayload()` to ensure godModeSuperUsers is always serialized as `Array<{ identifier: string }>` in the outgoing payload, never as `string[]`.

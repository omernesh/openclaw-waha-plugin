1. Create `src/admin/src/hooks/useTheme.ts`:
   ```typescript
   import { useState, useEffect } from 'react'

   const STORAGE_KEY = 'waha-admin-theme'

   export function useTheme() {
     const [theme, setTheme] = useState<'dark' | 'light'>(() => {
       return (localStorage.getItem(STORAGE_KEY) as 'dark' | 'light') ?? 'dark'
     })

     useEffect(() => {
       const root = document.documentElement
       root.classList.toggle('dark', theme === 'dark')
       root.classList.toggle('light', theme === 'light')
       localStorage.setItem(STORAGE_KEY, theme)
     }, [theme])

     const toggle = () => setTheme(t => (t === 'dark' ? 'light' : 'dark'))
     return { theme, toggle }
   }
   ```
   Key details:
   - Default is 'dark' (per locked decision)
   - useState initializer reads localStorage synchronously (before first render) to avoid flash
   - useEffect toggles BOTH 'dark' and 'light' classes on document.documentElement (html tag)
   - Persists to localStorage key `waha-admin-theme` (per locked decision)
